  var text = encodeURI("Hello world of translations, please translate me now!");

  if (fromTr !== ""){
      url += "&from=" + fromTr;
  }
  fetch(url, {
    method: 'get',
    headers: new Headers({ 'X-Preimage': preimage })
  }).then(function (response) {
    if (response.status === 200) {
      return response.json();
